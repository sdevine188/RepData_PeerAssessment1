impute missing values for steps using mean steps for that interval

steps.na <- which(is.na(data$steps) == TRUE)

data2 <- data

for(i in steps.na) {

        interval <- data$interval[i]

        row <- which(data.agg2$interval == interval)

        mean <- data.agg2$steps[row]

        data2$steps[i] <- mean
}
